157 
158  // dofs can be determined by checking the solutions' blocks an comparing the length of the vectors to the number of unique nodes.
159  // If the length is the same: dofs =1 , if it's twice as long: dofs =2 .. and so on
160

163  dofs_ = 1;
164  else if(2*(inputMeshP12_->getMapUnique()->getNodeNumElements()) == valuesSolution->getBlock(0)->getDataNonConst(0).size())
165  dofs_ = 2;

168

171  dofsP_ = 1;
